파일에 깃허브 연결 방법
# 예: Documents 폴더
cd C:\Users\USER\Documents
git clone https://github.com/DongyangOne/AInterview_front.git


npm install
-> 필요한 패키지 설치(초기에 해줘야됨)

git start
-> 코드 실행

git push
-> 코드 업로드

코드 업로드 예
git add . // 코드 업로드
git commit -m "Merge branch 'main' into front-9" // 커밋할 메시지 작성
git push origin main // 자신의 코드를 메인에 병합



git pull
-> 코드 최신 변경사항 가져오기

git pull origin front-9
